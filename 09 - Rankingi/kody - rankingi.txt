SELECT
	last_name,
	salary,
	row_number() over (order by salary) row_number
FROM
	employees

	
explain
select
	t1.*
	from (
		SELECT
			last_name,
			salary,
			row_number() over (order by salary) row_number
		FROM
			employees ) t1
where
	t1.row_number  < 4
	
	
	
SELECT
	last_name,
	salary,
	department_id,
	row_number() over (partition by department_id order by salary) row_number
FROM
	employees
	
	
SELECT
	last_name,
	salary,
	rank() over (order by salary) rnk_,
	dense_rank() over (order by salary) drnk_
FROM
	employees	
	
	
SELECT
	last_name,
	salary,
	rank() over (order by salary) rnk_,
	dense_rank() over (order by salary) drnk_
FROM
	employees	
	
	
select
	t1.*
	from (
		SELECT
			last_name,
			salary,
			department_id,
			rank() over (partition by department_id order by salary) rnk_
		FROM
			employees) t1
where
	t1.rnk_ < 4
		
select
	t1.val,
	cume_dist() over (order by t1.val) cd,
	percent_rank() over (order by t1.val) cd
	from(
		SELECT
			100 as val 
			union
			SELECT
			200 as val 
			union
			SELECT
			300 as val

	last_name,
	job_id,
	salary,
	ntile(4) over (order by salary ) ntile
FROM
	employees

	job_id in ('FI_ACCOUNT', 'IT_PROG')
