create or replace view v_dept_20
as
select
	e.last_name,
	e.salary,
	d.department_name,
	j.job_title
from employees e join departments d on e.department_id = d.department_id 
		join jobs j on e.job_id = j.job_id 
where e.department_id = 20



explain
select * from v_dept_20;

create materialized view mv_sum_dept
as
select
	d.department_name,
	sum(salary + (salary * coalesce(commission_pct,0))) dept_salary
from employees e join departments d on e.department_id = d.department_id 
group by d.department_name


explain
select * from mv_sum_dept;
