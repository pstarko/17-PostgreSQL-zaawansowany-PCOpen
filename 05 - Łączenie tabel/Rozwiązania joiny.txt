/*
1. Pokaż imię, nazwisko, numer działu i stanowisko dla wszystkich pracowników. 
Nie używaj konstrukcji „where department_id is null”, rozwiąż problem połączeniami tabel (106 wierszy)
*/

select
	e.first_name ,
	e.last_name ,
	d.department_id ,
	j.job_title 
	from employees e join departments d on e.department_id = d.department_id 
					 join jobs j on e.job_id = j.job_id 


/*
2. Pokaż pracowników pracujących w Seattle i Londynu (19 wierszy)
*/

select
	e.first_name ,
	e.last_name ,
	l.city 
	from employees e join departments d on e.department_id = d.department_id 
					 join locations l on l.location_id = d.location_id 
where
	l.city in ('Seattle', 'London')
	

/*
3. Pokaż poniższy raport nazwisk i nazw departamentów. Uwzględnij departamenty, które nie zatrudniają pracowników. 
Nie bierz pod uwagę zespołu Operations(121 wiersze)
*/

 select 
 	e.last_name, 
 	d.department_name  
 from employees e right join departments d on e.department_id=d.department_id 
 where d.department_name != 'Operations' 

/*
4. Pokaż ilość zatrudnionych pracowników w Stretford, Bombay, Sydney, London, Munich i Seattle(6 wierszy)
*/

select 
	l.city,
	count(e.employee_id ) ilosc_pracownikow
	from employees e join departments d on e.department_id =d.department_id
					 right join locations l on d.location_id = l.location_id 
where l.city in ('Stretford',
					'Bombay',
					'Sydney',
					'London',
					'Munich',
					'Seattle')
group by l.city
