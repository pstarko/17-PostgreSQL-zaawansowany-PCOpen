--1. Wyświetl nazwiska i roczne dochody pracowników(wraz z commission_pct) (107 wierszy)

select
	first_name,
	last_name,
	(salary + (salary * coalesce(commission_pct, 0))) * 12 roczna_placa
	from employees

--2.Wyświetl wszystkie informacje o departamentach sortując wynik według nazw (27 wierszy)

select
	*
	from departments
order by department_name
	
	
-- 3. Wyświetl listę job_id bez duplikatów(z tabeli employees) bez departamentu 20 i 50 (14 wierszy)

select
	distinct job_id
	from employees
where
	department_id not in (20, 50)
	
select
	distinct job_id
	from employees
where
	department_id <> 20
	and department_id <> 50


--4.Wyświetl employee_id, first_name, last_name, email i phone_number pracowników na etatach AC_ACCOUNT 
--i FI_ACCOUNT i posortuj ich według daty zatrudnienia(malejąco) (6 wierszy)

select 
	employee_id, 
	first_name, 
	last_name, 
	email,
	phone_number
	from employees
where
	job_id in ('AC_ACCOUNT', 'FI_ACCOUNT')
order by hire_date desc
	


-- 5.wybierz tylko employee_id, last_name i job_id z departamentów 30,40 w kolejności rosnących zarobków (7 wierszy)

select 
	employee_id, 
	last_name, 
	job_id
	from employees
where
	department_id in (30, 40)
order by salary

--6.Pokaż nazwiska i płace podstawowe(salary) tych, których płace podstawowe nie mieszczą się w przedziale 8000 do 15000 (74 wiersze)

select 
	last_name,
	salary
	from employees
where
	salary not between 8000 and 15000



-- 7.wyświetl nazwisko, etat(job_id) i numer departamentu tych których nazwisko kończy się na „an” (4 wiersze)

select 
	last_name,
	job_id,
	department_id
	from employees
where
	last_name like '%an'
