/*
1. Wyświetl nazwiska i stanowiska pracowników pracujących w tym samym dziale co pracownik o nazwisku Miller (5 wierszy)
*/

select
	first_name,
	last_name,
	(select job_title from jobs j where e.job_id = j.job_id ) jt
	from employees e
where
	department_id = (select department_id from employees
						where last_name = 'Miller')


/*
2. Wyświetl następujące dane o najdłużej zatrudnionym pracowniku na stanowisku FI_ACCOUNT (1 wiersz)
*/

select
	first_name ,
	last_name ,
	hire_date,
	job_id 
	from employees
where
	hire_date = (select min(hire_date) from employees
					where job_id = 'FI_ACCOUNT')
	and job_id = 'FI_ACCOUNT'
