/*
1. Wyświetl nazwiska i stanowiska pracowników pracujących w tym samym dziale co pracownik o nazwisku Miller (5 wierszy)
*/

select
	first_name,
	last_name,
	(select job_title from jobs j where e.job_id = j.job_id ) jt
	from employees e
where
	department_id = (select department_id from employees
						where last_name = 'Miller')


/*
2. Wyświetl następujące dane o najdłużej zatrudnionym pracowniku na stanowisku FI_ACCOUNT (1 wiersz)

	first_name ,
	last_name ,
	hire_date,
	job_id 
	from employees
where
	hire_date = (select min(hire_date) from employees
					where job_id = 'FI_ACCOUNT')

3. Zobacz najdłużej pracujących pracowników każdego działu. (12 wierszy)
*/
explain
select 	
	last_name,
	hire_Date,
	department_id,
	(select department_name from departments d where e.department_id = d.department_id ) dn
	from employees e
where 
	(department_id, hire_date) in (select department_id, min(hire_date) from employees
									group by department_id)

	(select department_name from departments d where e.department_id = d.department_id ) dn
	from employees e

	hire_date in (select min(hire_date) from employees

explain
select 	
	last_name,
	hire_Date,
	d.department_id,
	d.department_name
	from employees e join departments d on e.department_id = d.department_id
where 
	hire_date = (select min(hire_date) from employees
					where department_id = e.department_id )

/*
6. Wyświetl nazwiska poniższe dane pracowników związanych ze sprzedażą i mających
innego szefa niż John Singh – wykorzystaj w zapytaniu strukturę „derrived column”
(29 wierszy)
*/

select
	last_name,
	email,
	department_id,
	(select first_name || ' ' || last_name from employees where employee_id = e.manager_id ) boss
	from employees e
where
	job_id in (select job_id from jobs
				where lower(job_title) like '%sales%' )
	and manager_id != (select employee_id from employees
						where concat(first_name,' ' , last_name) = 'John Singh')

Użyj skorelowanego podzapytania, aby wyświetlić informacje o pracowniku niezatrudnionym w żadnym dziale (użyj właściwej formy operatora EXISTS) (1 wiersz)

8.
Wyświetl nazwę działu, który płaci Twoim pracownikom najwięcej pieniędzy miesięcznie(1 wiersz)

select
	*
	from employees e
where
	not exists (select 1 from departments d 
					where d.department_id = e.department_id )

select
	(select department_name from departments d where e.department_id = d.department_id)
	from employees e 	
group by e.department_id 
having sum(salary) = (
						select
							max(t1.suma)
							from (
								select
									department_id,
									sum(salary) suma
									from employees
