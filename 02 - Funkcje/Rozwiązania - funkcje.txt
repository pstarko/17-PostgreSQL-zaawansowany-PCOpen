--1. Dla każdego pracownika wygeneruj kod składający się z pierwszych dwóch liter job_id oraz całego employee_id

select
	last_name,
	job_id,
	left(job_id,2) || employee_id code
	from employees

--2. Wydaj wojnę literom l, m oraz k i zamień je na X. Użyj funkcji REPLACE i TRANSLATE.

select
	last_name,
	translate(last_name, 'lmk','XXX') translate_,
	replace(
		replace(
			replace(last_name, 'l', 'X')
								, 'm', 'X')
									, 'k', 'X') replace_
	from employees


--3. Dodaj podwyżkę 15% dla każdego pracownika. Uwzględnij również dodatek(commission_pct). 
-- Posortuj malejąco po podniesionej płacy.

select
	last_name,
	(salary + (salary * coalesce(commission_pct,0))) * 1.15 more_money
	from employees
order by more_money desc
