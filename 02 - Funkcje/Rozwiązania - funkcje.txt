--1. Dla każdego pracownika wygeneruj kod składający się z pierwszych dwóch liter job_id oraz całego employee_id

select
	last_name,
	job_id,
	left(job_id,2) || employee_id code
	from employees

--2. Wydaj wojnę literom l, m oraz k i zamień je na X. Użyj funkcji REPLACE i TRANSLATE.

select
	last_name,
	translate(last_name, 'lmk','XXX') translate_,
	replace(
		replace(
			replace(last_name, 'l', 'X')
								, 'm', 'X')
									, 'k', 'X') replace_
	from employees


--3. Dodaj podwyżkę 15% dla każdego pracownika. Uwzględnij również dodatek(commission_pct). 
-- Posortuj malejąco po podniesionej płacy.

select
	last_name,
	(salary + (salary * coalesce(commission_pct,0))) * 1.15 more_money
	from employees
order by more_money desc


--4.Policz ile dni jest zatrudniony każdy pracownik. Wyróżnij odpowiednim komentarzem tych, 
--którzy pracują ponad 3500 dni (Wynik na dzień 24-04-2025).

select
	last_name,
	now()::date teraz,
	current_date - hire_date różnica,
	case 
		when now()::date - hire_date > 3500 then 'Pracuje więcej niż 3,5k'
		else '----'
	end opis_1,	
	'2025-04-24'::date data,
	'2025-04-24'::date - hire_date różnica_2,
	case 
		when '2025-04-24'::date - hire_date > 3500 then 'Pracuje więcej niż 3,5k'
		else '----'
	end opis_1
	from employees

--5. Rozwiąż wyrażenie algebraiczne: Resztę z dzielenia 134 przez 8 przemnóż przez pierwiastek z 81. 
-- Podziel to wszystko przez 244 i podnieś do 4 potęgi. 
--Do wyniku dodaj 0.0097. Zaogrąglij wszystko do 4 miejsc po przecinku zgodnie z zasadami matematyki.


select
round((power(134%8 * sqrt(81) / 244, 4) + 0.0097)::numeric, 4)


--6. Pokaż last name w taki sposób, by ostatnie trzy litery były w odwróconej kolejności


select
	last_name,
	length(last_name),
	substring(last_name, 1, case 
								when length(last_name) - 3 < 0 then 0 
								else length(last_name) - 3
								end),
	substring(last_name, 1, case 
								when length(last_name) - 3 < 0 then 0 
								else length(last_name) - 3
								end) ||
	reverse(	right(last_name, 3)) code
	from employees


-- 6. Wiedząc, że do premii trzeba pracować 10 lat pokaż dane: datę zatrudnienia, ile lat
-- pracy, ile lat do emerytury oraz informacje o tym, że ktoś ma premię. Stan na 24-04-
-- 2025.


select 
	last_name,
	hire_date,
	age('2025-04-24', hire_date),
	date_part('year', age('2025-04-24', hire_date)) ilosc_lat,
	10 - date_part('year', age('2025-04-24', hire_date)) ile_do_premii,
	case 
		when 10 - date_part('year', age('2025-04-24', hire_date)) <0 then 'premiowany'
		when 10 - date_part('year', age('2025-04-24', hire_date)) = 0 then 'premia w tym roku'
		else 'jeszcze ' || 10 - date_part('year', age('2025-04-24', hire_date)) || 'lat/a'
	end result	
	from employees
	

	
create view query_1 as
select 
	last_name,
	hire_date,
	date_part('year', age('2025-04-24', hire_date)) ilosc_lat,
	10 - date_part('year', age('2025-04-24', hire_date)) ile_do_premii
	from employees
	

select 
	q.*,
	case 
		when ile_do_premii <0 then 'premiowany'
		when ile_do_premii = 0 then 'premia w tym roku'
		else 'jeszcze ' || ile_do_premii || 'lat/a'
	end result	
	from query_1	 q
